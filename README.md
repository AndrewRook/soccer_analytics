$ conda create -n soccer_analytics -c conda-forge -c fastai -c pytorch python=3.11 pytorch==2.1.0 fastai==2.7.13 xgboost==1.7.6 scikit-learn==1.3.0 pandas==2.1.3 notebook -y
$ conda activate soccer_analytics
$ pip install -e .
